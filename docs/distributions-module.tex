            {\it (type=int)}

          \item[p]


Probability of each one of the different outcomes,
$\sum_{i=1}^k p_i = 1)$, $p_i \ge 0$.
            {\it (type=(k,) float)}

        \end{Ventry}

      \end{quote}

\textbf{Note:} \begin{itemize}
